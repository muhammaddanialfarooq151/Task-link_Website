  <script>
    // Scroll animation
    const steps = document.querySelectorAll('.step');
    window.addEventListener('scroll', () => {
      const triggerBottom = window.innerHeight / 5 * 4;
      steps.forEach(step => {
        if (step.getBoundingClientRect().top < triggerBottom) {
          step.classList.add('visible');
        }
      });
    });

    const displayTitles = {
      "Home & Maintenance": "Skilled Service Providers",
      "Technology Services": "IT & Tech Experts"
    };

    const professionalsData = {
      "Home & Maintenance": [
        { name: "Ali Khan", degree: "Diploma in Carpentry", experience: "5 Years", contact: "03211234567" }
      ],
      "Technology Services": [
        { name: "Danial Farooq", degree: "B.Sc Computer Science", experience: "4 Years", contact: "03287602950" }
      ]
    };

    function showProfessionals(category) {
      const container = document.getElementById('professionalsContainer');
      container.innerHTML = `<h2>${displayTitles[category]}</h2>`;

      professionalsData[category].forEach(p => {
        container.innerHTML += `
          <div class="professional-card">
            <h3>${p.name}</h3>
            <p>${p.degree}</p>
            <p>${p.experience}</p>
            <p><a href="tel:${p.contact}">${p.contact}</a></p>
          </div>`;
      });

      container.style.display = "block";
      container.scrollIntoView({ behavior: 'smooth' });
    }
  </script>

</body>
</html>
